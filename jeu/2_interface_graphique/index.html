Création de la fenêtre mère

routine de démarrage
app = QApplication(sys.argv)#on créé la routine de démarrage propre à PySide6
def run(n):
    global run,window #on passe en global la fonction run et la variable window
    if n == 1:#lorsqu'on veut rejouer : run(1)
        del window #on supprime la fenêtre déjà existante
    window = MainWindow()#on créé une fenêtre
    window.show()#on la montre
    if n == 0:#lors d'un simple démarrage : le jeu est lancé via le fichier
        window.messagebox()#on montre la box de message contenant les règles pour jouer
        app.exec()#on lance la routine de démarrage
  • pour n == 0 : c'est un simple démarrage (quand on lance le jeu depuis le fichier) et on créé la fenêtre : window = MainWindow()
  • pour n == 1 : c'est quand on veut rejouer : on supprime la fenêtre de jeu, et on relance la fonction : elle re-setUp la fenêtre

def Setup(self): #Set up des compartiments dans les différents compartiments
    #---------------Mise en place d'un Compartiment dans la partie centrale
    self.Frame = QFrame()
    self.Frame.setAttribute(Qt.WA_StyledBackground, True)
    self.Frame.setStyleSheet("background-color : #777777")
    self.Frame.setContentsMargins(0,0,0,0)
    self.setCentralWidget(self.Frame)# Compartiment parent
    # Creation des layout : conteneurs des widgets dans les compartiments
    self.GridFrame = QGridLayout(self.Frame)
    #---------------Compartiments enfants de self.GridFrame---------------------
    #---------------- Récipient Couleurs ---------------------------------------
    self.dessin = QWidget()
    self.dessin.setFixedSize(510,620)
    self.dessin.setStyleSheet("background : #444444;")
    self.GridFrame.addWidget(self.dessin,0,0)
    #---------------- Elements additionnels ------------------------------------
    self.WidgetDroite = QWidget()
    self.WidgetDroite.setFixedSize(490,620)
    self.WidgetDroite.setStyleSheet("background : #444444; color : white")
    self.GridFrame.addWidget(self.WidgetDroite,0,1)
    #---------------Mise en place d'un Compartiment dans la barre de statut
    Status = self.statusBar()# Compartiment parent défini en même temps que la création
    Status.setStyleSheet("background : red; color : white")# Style perso
    Status.showMessage("Ceci est un Mastermind!")# Message
    #---------------Mise en place d'un Compartiment dans le menu
    self.Menu = QWidget()
    self.Menu.setStyleSheet("background : black; color : white")# Style perso
    self.setMenuWidget(self.Menu)# Compartiment parent
    self.Menu.setFixedHeight(50)# Taille
On remarque :

  • self.Frame est une QFrame placée dans le Widget Central qui prend l'espace disponible,
    on active sa fonction de coloriage et on la colorie en gris
  • self.GridFrame est une QGridFrame ayant pour parent self.Frame donc elle est automatiquement mise dans QFrame
    C'est dans cette GridFrame qu'on va organiser tous les sous éléments (= les nouveaux conteneurs)
  • self.dessin est un QWidget ayant une taille de 510 sur 620 px avec une couleur de fond gris foncée
    Par la suite self.dessin est mis dans self.GridFrame à la colonne 0 (la 1ère) et ligne 0 (la 1ère) self.dessin accueillera les ronds servant aux tentatives et les indications sur ces dernières
  • self.WidgetDroite est un QWidget ayant une taille de 490 sur 620 px avec la même couleur de fond que self.dessin Ce Widget se situe dans self.GridFrame à la colonne 1 et la ligne 0 self.WidgetDroite sert à mettre la palette de couleur pour choisir et les boutons pour mettre la couleur dans la case souhaitée
  • Status est la Barre de Statut, sa couleur de fond est rouge et celle d'écriture blanche, on met un message informatif
  • self.Menu est un QWidget situé dans le Widget de Menu ayant une couleur de fond noire et une couleur d'écriture blanche
    Sa hauteur est de 50

class Button(QPushButton):#Classe qui créé des boutons et qui hérite de QPushButton
    def __init__(self,texte):#Initialisation
        super().__init__()#On recupère les propriétés de QPushButton
        self.texte = int(texte)#On recupère le Str passé en paramètre lors de la création en ints
        self.setFixedSize(75,30)#Taille fixé
        self.button_is_checked = True#On définit il est enfoncé
        self.setCheckable(True)#On définit qu'il peut-être enfoncé
        self.setText(texte)#On définit son texte par le Str passé en paramètre lors de la création
        self.clicked.connect(self.the_button_was_toggled)#Au click on appelle une fonction
        self.setStyleSheet("background-color: black;")#On définit du style

    def the_button_was_toggled(self):#Lorsqu'on clique
        self.setChecked(True)#On enfonce le bouton
        print(f"Bouton sélectionné : {self.texte}")#Debug
        window.toggle(self.texte)#On appelle une fonction qui sert à désenfoncer les autres boutons : un à la fois enfoncé
        window.bouton_locked = self.texte#bouton_locked prend la valeur du bouton enfoncé : self.texte

    def toggled(self):#lors de l'appel on desenfonce le bouton
        self.setChecked(False)

class Button_principaux(QPushButton):#Classe créant des boutons hériant de QPushButton

        elif self.text() == "Comparer!":# SI le bouton est => Comparer!
            window.compare()#On appelle la fonction de MainWindow pour comparer 

    def disable(self):#On désactive et on met aucun texte : ""
        self.setDisabled(True)
        self.setText("")
    def enable(self):#On active le bouton
        self.setEnabled(True)

class Indication(QWidget):#héritant de QWidget utilisant QPainter pour dessiner
    def __init__(self):#Initialisation
        super().__init__()#Héritage
        self.liste_couleur = []# même nombre de boutons que de couleurs donc une seule liste
    def paintEvent(self,event):#Evenement dessin
        #Routine pour dessiner comme dans Draw()
        paint = QtGui.QPainter(self)
        pen = QtGui.QPen(QtGui.QColor(68,68,68))#Couleur de fond : gris
        pen.setWidth(1)#Taille
        paint.setPen(pen)
        for n in range(len(self.liste_couleur)):#On parcourt la longueur de la liste en fonction
            #du nombre de ligne de points à mettre
            for i in range(len(self.liste_couleur[n])):
                #On parcourt la longueur de la liste en fonction
                #du nombre de points à mettre
                paint.setBrush(self.liste_couleur[n][i])#Couleur correspondant 
                paint.drawEllipse(10*(i+1)+10*i,60*n+30,10,10)#Dessin du cercle

def compare(self):
    tentative = self.draw.liste_code[self.tour]#on recupère la liste à la position self.tour
    # dans l'instance de Draw()
    choix = self.return_tirage#le tirage fait dans la fonction tirage()
    print(f"Tour = {self.tour}")#Debug
    print(f"Choix = {choix}")#Debug
    #------------- Creation des messages d'erreurs
    erreur_cercles = QMessageBox(self)
    erreur_cercles.setWindowTitle("Alert")
    erreur_cercles.setText("Veuillez sélectionner une couleur sur toutes les cases!")
    erreur_couleurs = QMessageBox(self)
    erreur_couleurs.setWindowTitle("Alert")
    erreur_couleurs.setText("Vous ne pouvez pas sélectionner 2 fois une même couleur!")
    place = 0#nombre de chiffres bien placés
    trouve = 0#nombre de chiffres trouvés
    stop = 0#cas d'arrête si stop > 0
    for elt in tentative:#on cherche si dans tentative 
        #il y a une case sans code = "Faux"
        if elt == "Faux":
            stop += 1
    if stop == 0: #si pas d'erreur detecté
        print(f"tentative : {tentative}")#Debug
        for n in range(4):# donc on sait que tentative a une longueur de 4 dans tous les cas => range(4)
            #on cherche à savoir si 2 éléments sont identiques
            for k in range(4):
                if k != n and tentative[n] == tentative[k]:
                    stop = 5
                    print(f"stop : {stop}")
    if stop == 0:#si pas d'erreur
        if choix == tentative:#le choix est identique donc, tout est trouvé
            self.win()#cas de victoire
        else:#sinon on cherche le nombre de place et trouve
            for i in range(len(choix)):
                for y in range(len(tentative)):
                    if y !=None:
                        if tentative[y] == choix[i]:#chiffre identique
                            trouve = trouve + 1
                            if i == y:#bonne position
                                place = place + 1
                                trouve = trouve - 1
            if self.tour == 9:# si c'est le dernier tour
                self.loose()
            else:#sinon 
                self.place_trouve(trouve,place)#fonction pour dessiner les indications
                print(f"Trouve = {trouve} place = {place}")#Debug
    elif stop > 0:#En cas d'erreur
        self.tour -= 1#On reste au même tour car +1 -1
        print(f"Tour enlevé de 1 = {self.tour}")#Debug
        if stop < 5:#Selon la valeur de stop : différente erreur
            erreur_cercles.exec()
        if stop == 5:
            erreur_couleurs.exec()
    self.tour += 1

def place_trouve(self,trouve,place):#pour les indications
    liste = []#liste qui sera ajouté à celle de la classe Indication()
    for i in range(trouve):#ajout du vert
        liste.append(Qt.green)
    for i in range(place):#ajout du rouge
        liste.append(Qt.red)
    if liste == []:#si liste vide on créé un point invisible car couleur de fond
        liste = [QtGui.QColor(68,68,68)]
    print(liste)#Debug
    self.indication.liste_couleur.insert(len(self.indication.liste_couleur),liste)#ajout de liste à la dernière place
    self.indication.repaint()#repaint
    print(self.indication.liste_couleur)#Debug

def win(self):#cas de victoire
    message = QMessageBox(self)#Message de victoire
    message.setWindowTitle("Gagné")
    message.setText("Bien joué vous avez trouvé les bons chiffres, vous avez gagné!")
    message.exec()#executé
    self.comparer.disable()#bouton désactivé
    self.jouer.enable()#bouton activé
    self.jouer.setText("Rejouer!")#titre défini

def loose(self):#cas de défaite

    message.setWindowTitle("Perdu")
    message.setText("Dommage pour vous, vous n'avez pas réussi à trouver le chiffre, il ne vous reste plus de coup vous avez perdu")
    message.exec()#executé
    #----Même routine que pour le cas de victoire
    self.comparer.disable()
    self.jouer.enable()
    self.jouer.setText("Rejouer!")

def palette(self):#Fonction qui créé la palette de couleurs
    self.Widget = QWidget()#On créé le Wigdet qui va accueillir la palette
    self.Widget.setStyleSheet("background-color : #444444")#On définit une couleur de fond
    self.Widget.setGeometry(340, 0,320,225)#Set position du widget et la taille,
    #on donne que l'éloignement par rapport à gauche car celui par rapport au haut est fait par le layout
    self.color = ['white','#c0c0c0','#111111','black','#800080', '#000080','blue','#007fff','#32cd32',
    '#7cfc00','yellow','#ffb800', '#ff4500','#ff1493', '#801000','red']# liste de couleur pour la palette
    layout_couleurs = QGridLayout()#Grille qui va accueillir les nouveaux layouts
    layout_couleurs.setContentsMargins(6,5,6,5)#On définit une marge pour chaque élément
    layout_couleurs.setHorizontalSpacing(30)#Espacement horizontal
    layout_1 = QVBoxLayout()#QVBoxLayout créé
    layout_1.setSpacing(3)#Espacement set
    layout_2 = QVBoxLayout()#QVBoxLayout créé
    layout_2.setSpacing(3)#Espacement set
    layout_3 = QVBoxLayout()#QVBoxLayout créé
    layout_3.setSpacing(3)#Espacement set
    layout_4 = QVBoxLayout()#QVBoxLayout créé
    layout_4.setSpacing(3)#Espacement set
    for i in range(16):#16 couleurs donc on fait une boucle
        elt = MyWidget(i)#Pour 16 fois, elt prend la valeur MyWidget() avec en paramètre la valeur de i
        couleur = self.color[i]#Pour 16 fois, couleur prend la valeur self.color[i]
        elt.setAttribute(Qt.WA_StyledBackground, True)#Pour chaque element on active la couleur de fond
        elt.setStyleSheet(f"background-color : {couleur}")#Style
        if i > 11:#en fonction de i on le place dans un layout différent
            layout_4.addWidget(elt)
        elif i > 7:
            layout_3.addWidget(elt)
        elif i > 3:
            layout_2.addWidget(elt)
        else:
            layout_1.addWidget(elt)
        elt.setFixedSize(50,50)#pour chzque elt, on lui donne une taille fixe
    print("Palette set!")#Debug
    layout_couleurs.addLayout(layout_1,0,0)#On ajoute au layout_couleur chaque layout créé
    layout_couleurs.addLayout(layout_2,0,1)
    layout_couleurs.addLayout(layout_3,0,2)
    layout_couleurs.addLayout(layout_4,0,3)
    self.Widget.setLayout(layout_couleurs)#On ajoute le layout_couleur au Widget récipient. 
    #Le Wigdet sera ajouté après dans la fonction layout_WidgetDroite()

        #En lien avec les ronds récipients : abordé en haut
        window.draw.liste_color[window.tour][window.bouton_locked] = window.draw.liste_color_palette[self.souris]

        # au dessus mais avec un code
